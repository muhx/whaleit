       ```tsx
       import { ArrowDownLeft, ArrowUpRight, ArrowLeftRight, MessageSquare, SplitSquareHorizontal } from "lucide-react";
       import { Tag } from "lucide-react";
       import { cn } from "@/lib/utils";
       import { PrivacyAmount } from "@whaleit/ui";
       import type { Transaction, TransactionDirection } from "@/lib/types/transaction";

       export interface TransactionRowProps {
           transaction: Transaction;
           variant?: "default" | "account-suppressed";
           showRunningBalance?: boolean;
           runningBalance?: number | null;
           baseCurrency: string;
           accountName?: string;          // resolved by parent
           categoryName?: string;
           categoryColor?: string;
           onClick?: () => void;
       }

       const ICONS: Record<TransactionDirection, { icon: React.ComponentType<any>; className: string }> = {
           INCOME: { icon: ArrowDownLeft, className: "text-success" },
           EXPENSE: { icon: ArrowUpRight, className: "text-muted-foreground" },
           TRANSFER: { icon: ArrowLeftRight, className: "text-muted-foreground" },
       };

       export function TransactionRow({ transaction, variant = "default", showRunningBalance = false, runningBalance, baseCurrency, accountName, categoryName, categoryColor, onClick }: TransactionRowProps) {
           const { icon: Icon, className: iconClass } = ICONS[transaction.direction];
           const showFx = transaction.currency !== baseCurrency && transaction.fxRate != null;
           const baseEquivalent = showFx ? transaction.amount * (transaction.fxRate ?? 1) : null;
           const sign = transaction.direction === "INCOME" ? "+" : transaction.direction === "EXPENSE" ? "−" : "";
           const amountClass = transaction.direction === "INCOME" ? "text-success" : "text-foreground";

           return (
               <button
                   type="button"
                   onClick={onClick}
                   className="flex w-full items-center gap-3 px-4 py-3 hover:bg-card/50 transition-colors duration-150 text-left"
                   data-testid="transaction-row"
                   aria-label={`Transaction: ${transaction.payee ?? "Transfer"} ${sign}${transaction.amount} ${transaction.currency}`}
               >
                   <Icon className={cn("size-4", iconClass)} aria-hidden="true" />
                   <div className="flex-1 min-w-0">
                       <div className="flex items-center gap-1.5">
                           <span className="font-semibold text-base truncate">{transaction.payee ?? "Transfer"}</span>
                           {transaction.notes && (
                               <MessageSquare className="size-3 text-muted-foreground/60 shrink-0" aria-label="has note" />
                           )}
                       </div>
                       <div className="text-xs text-muted-foreground flex items-center gap-1.5">
                           {transaction.hasSplits ? (
                               <SplitBadge count={transaction.splits.length} />
                           ) : (
                               <CategoryChip name={categoryName} color={categoryColor} />
                           )}
                           {variant !== "account-suppressed" && accountName && <><span aria-hidden="true">·</span><span>{accountName}</span></>}
                           <span aria-hidden="true">·</span>
                           <span>{transaction.currency}</span>
                       </div>
                   </div>
                   <div className="text-right shrink-0">
                       <div className={cn("font-semibold tabular-nums text-base", amountClass)}>
                           {sign}<PrivacyAmount value={transaction.amount} currency={transaction.currency} />
                       </div>
                       {showFx && (
                           <div className="text-xs text-muted-foreground tabular-nums">
                               ~<PrivacyAmount value={baseEquivalent ?? 0} currency={baseCurrency} />
                           </div>
                       )}
                       {showRunningBalance && runningBalance != null && (
                           <div className="text-xs text-muted-foreground tabular-nums">
                               Bal <PrivacyAmount value={runningBalance} currency={transaction.currency} />
                           </div>
                       )}
                   </div>
               </button>
           );
       }
       ```
