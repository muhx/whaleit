           let parent_db: TransactionDB = new.clone().into();
           diesel::insert_into(transactions::table).values(&parent_db).execute(conn).await?;
           if !new.splits.is_empty() {
               let split_dbs: Vec<TransactionSplitDB> = new.splits.iter()
                   .enumerate()
                   .map(|(i, s)| TransactionSplitDB {
                       id: Uuid::now_v7().to_string(),
                       transaction_id: parent_db.id.clone(),
                       category_id: s.category_id.clone(),
                       amount: s.amount,
                       notes: s.notes.clone(),
                       sort_order: i32::try_from(i).unwrap_or(0),
                       created_at: Utc::now().naive_utc(),
                       updated_at: Utc::now().naive_utc(),
                   })
                   .collect();
               diesel::insert_into(transaction_splits::table).values(&split_dbs).execute(conn).await?;
           }
           Ok(())
       }.scope_boxed()).await?;

       self.get_by_id(&parent_db.id).await
