       b. **`import_csv`** — Multipart handler:
       ```rust
       async fn import_csv(
           State(state): State<AppState>,
           AuthUser(_user): AuthUser,
           mut multipart: Multipart,
       ) -> Result<Json<ImportResult>, ApiError> {
           let mut account_id = String::new();
           let mut file_bytes = Vec::new();
           let mut mapping_json = String::new();
           let mut template_name: Option<String> = None;
           while let Some(field) = multipart.next_field().await? {
               let name = field.name().unwrap_or("").to_string();
               match name.as_str() {
                   "accountId" => account_id = field.text().await?,
                   "file" => {
                       let bytes = field.bytes().await?;
                       if bytes.len() > MAX_IMPORT_BYTES {
                           return Err(ApiError::PayloadTooLarge);
                       }
                       file_bytes = bytes.to_vec();
                   }
                   "mapping" => mapping_json = field.text().await?,
                   "templateName" => template_name = Some(field.text().await?),
                   _ => {}
               }
           }
           let mapping = serde_json::from_str(&mapping_json)?;
           let req = CoreCsvImportRequest { account_id, file_bytes, mapping, template_name };
           let result = state.transaction_service.import_csv(req).await?;
           Ok(Json(result))
       }
       ```
