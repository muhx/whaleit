    ```typescript
    import { expect, Page, test } from "@playwright/test";
    import { BASE_URL, loginIfNeeded } from "./helpers";

    test.describe.configure({ mode: "serial" });

    const checkingName = `E2E Checking ${Date.now()}`;
    const ccName = `E2E CreditCard ${Date.now()}`;

    test.describe("Phase 3 — Bank Accounts & Credit Cards", () => {
      let page: Page;

      test.beforeAll(async ({ browser }) => {
        page = await browser.newPage();
      });

      test.afterAll(async () => {
        await page.close();
      });

      test("1. Setup: login + open /settings/accounts", async () => {
        test.setTimeout(180_000);
        await loginIfNeeded(page);
        await page.goto(`${BASE_URL}/settings/accounts`, { waitUntil: "domcontentloaded" });
        await expect(page.getByRole("heading", { name: "Accounts", exact: true })).toBeVisible();
      });

      test("2. Create CHECKING account", async () => {
        // Open the new-account flow. Use the same trigger as the existing form.
        const addButton = page.getByRole("button", { name: /Add account/i });
        await expect(addButton).toBeVisible();
        await addButton.click();

        // Modal opens. Fill name, select CHECKING, fill institution, fill opening balance, currency.
        await page.getByLabel(/Account Name/i).fill(checkingName);

        // Select CHECKING in the type picker. Be tolerant of either ResponsiveSelect or ToggleGroup.
        const typeTrigger = page.getByLabel(/Account Type/i);
        await typeTrigger.click();
        await page.getByRole("option", { name: /Checking/i }).click().catch(async () => {
          // Fallback: ToggleGroup with Checking label
          await page.getByRole("button", { name: /^Checking$/i }).click();
        });

        // Institution
        await page.getByLabel(/Institution/i).fill("Test Bank");

        // Opening balance — MoneyInput. Try by label, fall back to first input near the label.
        const openingBalance = page.getByLabel(/Opening balance/i);
        await openingBalance.fill("0");

        // Submit
        const submitButton = page.getByRole("button", { name: /Create account|Save/i });
        await submitButton.click();

        // Verify the row appears
        await expect(page.getByText(checkingName)).toBeVisible({ timeout: 10_000 });
      });

      test("3. Create CREDIT_CARD account with required CC fields", async () => {
        const addButton = page.getByRole("button", { name: /Add account/i });
        await addButton.click();

        await page.getByLabel(/Account Name/i).fill(ccName);

        const typeTrigger = page.getByLabel(/Account Type/i);
        await typeTrigger.click();
        await page.getByRole("option", { name: /Credit Card/i }).click().catch(async () => {
          await page.getByRole("button", { name: /^Credit Card$/i }).click();
        });

        await page.getByLabel(/Institution/i).fill("Test Bank");
        await page.getByLabel(/Opening balance/i).fill("0");

        // CC fields — should now render
        const creditLimit = page.getByLabel(/Credit limit/i);
        await expect(creditLimit).toBeVisible();
        await creditLimit.fill("5000");

        const cycleDay = page.getByLabel(/Statement cycle day/i);
        await cycleDay.click();
        await page.getByRole("option", { name: "15" }).click().catch(async () => {
          // Fallback: a generic select
          await cycleDay.fill("15");
        });

        const submitButton = page.getByRole("button", { name: /Create account|Save/i });
        await submitButton.click();

        await expect(page.getByText(ccName)).toBeVisible({ timeout: 10_000 });
        // Available credit chip should appear on the CC row
        await expect(page.getByText(/Available/i).first()).toBeVisible();
      });

      test("4. Update balance via modal", async () => {
        // Click into the CC account.
        await page.getByText(ccName).click();

        // Click the Update balance button on the detail page.
        const updateBtn = page.getByRole("button", { name: /Update balance/i });
        await expect(updateBtn).toBeVisible();
        await updateBtn.click();

        // Modal opens. Fill new balance and save.
        const newBalance = page.getByLabel(/New balance/i).or(page.locator("input[inputmode='decimal']").first());
        await newBalance.fill("750");

        const saveButton = page.getByRole("button", { name: /Save balance/i });
        await saveButton.click();

        // Toast should appear, modal should close.
        await expect(page.getByText(/Balance updated just now/i)).toBeVisible({ timeout: 5_000 });
      });

      test("5. Archive account, confirm hidden by default", async () => {
        await page.goto(`${BASE_URL}/settings/accounts`, { waitUntil: "domcontentloaded" });

        // Find the checking row and trigger its archive action via the row's overflow menu.
        const checkingRow = page.getByText(checkingName);
        await expect(checkingRow).toBeVisible();

        // The exact archive trigger depends on the row's actions widget.
        // Try a few common patterns.
        await checkingRow.locator("..").getByRole("button", { name: /options|more|menu/i }).first().click().catch(() => {});

        await page.getByRole("menuitem", { name: /Archive/i }).first().click().catch(async () => {
          // Fallback: navigate to the account page and archive there
          await checkingRow.click();
          await page.getByRole("button", { name: /Archive/i }).click();
          await page.getByRole("button", { name: /^Archive$/ }).click();
        });

        // Back to the list — verify the row is no longer visible (default = hide archived)
        await page.goto(`${BASE_URL}/settings/accounts`, { waitUntil: "domcontentloaded" });
        await expect(page.getByText(checkingName)).toHaveCount(0);
      });

      test("6. Show archived toggle reveals archived row", async () => {
        await page.goto(`${BASE_URL}/settings/accounts`, { waitUntil: "domcontentloaded" });

        // Flip the Show archived Switch.
        const showArchived = page.getByLabel(/Show archived/i).or(page.getByRole("switch", { name: /Show archived/i }));
        await showArchived.click();

        // Archived row should now be visible.
        await expect(page.getByText(checkingName)).toBeVisible({ timeout: 5_000 });
      });
    });
    ```

    NOTE: Some selectors are deliberately tolerant (using `.or()` and `.catch()`)
    because the exact widget — ResponsiveSelect vs ToggleGroup, options vs
    menuitem — depends on the executor's choices in Plan 03-07. The spec is
    written to flex around reasonable variations. If the spec fails because of
    a label mismatch, the executor adjusts the selector (preferred) or aligns
    the page to UI-SPEC labels.
